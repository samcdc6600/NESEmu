159 /* Throws exception if number is greater then
160  "std::numeric_limits<unsigned char>::max()" or
161  "std::numeric_limits<char>::max()", depending on whether T is an unsigned
162  char or a char, respectively. */
163 template <typename T> void checkForCharNumRange(size_t number,
164  const size_t callCount)
165 // requires (typeid(T) == typeid(unsigned char) || typeid(T) == typeid(char))
166 {
167  std::stringstream e {};
168  // Replace with constrint in the future?
169  assert(typeid(T) == typeid(unsigned char) || typeid(T) == typeid(char)
170  && "Error T != unsigned char or T != char!\n");
171 
172  if(typeid(T) == typeid(unsigned char))
173  {
174  if(number > std::numeric_limits<unsigned char>::max())
175  {
176  e<<"argument #"<<callCount<<" invalid";
177  throw std::invalid_argument(e.str());
178  }
179  }
180  else
181  {
182  if(typeid(T) == typeid(char))
183  {
184  if(number > std::numeric_limits<char>::max())

209 /* Extracts numbers of type T for str and stores in retNumber, throws exception
210  if there number cannot be read from str or if there are too few (a case where
211  the number cannot be read) or too many numbers in str. Note that the argument
212  callCount should generally be 0 unless you are doing something wired. */
213 template <typename T, typename ... Args>
214 void getNumbersFromStrInHex(std::stringstream & str, size_t callCount, T & retNumber,
215  Args & ... args) // requires std::is_integral<T>::value
216 {
217  callCount++;
218  /* We have to make sure that number isn't of type "char" or "unsigned char"
219  beacuse it will only read in one character from the stream instead of the
220  desired two. */
221  if(isNumChar(retNumber))
222  {
223  size_t nonCharNum {};
224  str>>std::hex>>nonCharNum;
225  checkForCharNumRange<T>(nonCharNum, callCount);
226  retNumber = nonCharNum;
227  }
228  else
229  {
230  str>>std::hex>>retNumber;
231  }
232 
233  if(str.fail())
234  {
235  std::stringstream e {};
236  if(str.eof())
237  {
238  e<<"too few arguments read "<<(callCount -1)<<" arguments, or "
239  "arguments too long";
240  throw std::invalid_argument(e.str());
241  }
242  else
243  {
244  e<<"argument #"<<callCount<<" invalid";
245  throw std::invalid_argument(e.str());
246  }
247  }
248  else
249  {
250  getNumbersFromStrInHex(str, callCount, args...);
251  }
252 }

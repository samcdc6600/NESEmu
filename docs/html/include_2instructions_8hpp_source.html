178 inline void bpl_10()
179 { /* Branches are dependant on the status of the flag bits when the op code is
180  encountered. A branch not taken requires two machine cycles. Add one if the
181  branch is taken and add one more if the branch crosses a page boundary. */
182  if(architecturalState::status.u.N == 0)
183  branchTaken();
184  architecturalState::PC += 2; // This is done even if branch is taken.
185  architecturalState::cycles += 2;
186 }

321 inline void cmp_c9()
322 { /* Compare sets flags as if a subtraction had been carried out. If the value
323  in the accumulator is equal or greater than the compared value, the Carry
324  will be set. The equal (Z) and negative (N) flags will be set based on
325  equality or lack thereof and the sign (i.e. A>=$80) of the accumulator.
326  - http://6502.org/tutorials/6502opcodes.html#CMP */
327  architecturalState::status.u.C = ((architecturalState::A == get8BitImmediate())
328  || (architecturalState::A >
329  get8BitImmediate())) ? 1 : 0;
330  setZeroFlagOn(architecturalState::A - get8BitImmediate());
331  setNegativeFlagOn(architecturalState::A - get8BitImmediate());
332  architecturalState::PC += 2;
333  architecturalState::cycles += 2;
334 }
